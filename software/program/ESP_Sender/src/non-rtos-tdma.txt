#include "main.h"
#include <math.h>

#define LOG_SIZE 1000
#define PACKET_TOTAL_LEN (SECURE_DATA_TOTAL_LEN + sizeof(uint16_t))

// ==================== CONFIGURATION ====================
#define TX_INTERVAL_MS      100     // Transmit every 100ms (10 Hz)
#define SENSOR_SAMPLE_MS    10      // Sensor sampling rate (100 Hz)

// ==================== NODE ID ====================
volatile uint8_t u8SlaveNodeID = 0;
volatile uint16_t g_seq_id = 0;

// ==================== LATENCY STATISTICS ====================
typedef struct {
    uint32_t count;
    float    min_us;
    float    max_us;
    float    sum;
    float    sum_sq;
    float    last_value;
    float    jitter_sum;
    uint32_t jitter_count;
} LatencyStats;

typedef struct {
    LatencyStats sensor_read;
    LatencyStats encrypt;
    LatencyStats lora_tx;
    LatencyStats total_cycle;   // sensor -> encrypt -> TX complete
    LatencyStats loop_time;
} SystemLatencyStats;

SystemLatencyStats g_latency_stats;

void latency_stats_init(LatencyStats* stats) {
    stats->count = 0;
    stats->min_us = 999999999.0f;
    stats->max_us = 0.0f;
    stats->sum = 0.0f;
    stats->sum_sq = 0.0f;
    stats->last_value = 0.0f;
    stats->jitter_sum = 0.0f;
    stats->jitter_count = 0;
}

void latency_stats_update(LatencyStats* stats, float value_us) {
    stats->count++;
    if (value_us < stats->min_us) stats->min_us = value_us;
    if (value_us > stats->max_us) stats->max_us = value_us;
    stats->sum += value_us;
    stats->sum_sq += (value_us * value_us);
    if (stats->count > 1) {
        stats->jitter_sum += fabs(value_us - stats->last_value);
        stats->jitter_count++;
    }
    stats->last_value = value_us;
}

float latency_stats_mean(LatencyStats* stats) {
    if (stats->count == 0) return 0.0f;
    return stats->sum / stats->count;
}

float latency_stats_variance(LatencyStats* stats) {
    if (stats->count < 2) return 0.0f;
    float mean = latency_stats_mean(stats);
    return (stats->sum_sq / stats->count) - (mean * mean);
}

float latency_stats_std(LatencyStats* stats) {
    float var = latency_stats_variance(stats);
    return (var > 0) ? sqrtf(var) : 0.0f;
}

float latency_stats_jitter(LatencyStats* stats) {
    if (stats->jitter_count == 0) return 0.0f;
    return stats->jitter_sum / stats->jitter_count;
}

void system_latency_init() {
    latency_stats_init(&g_latency_stats.sensor_read);
    latency_stats_init(&g_latency_stats.encrypt);
    latency_stats_init(&g_latency_stats.lora_tx);
    latency_stats_init(&g_latency_stats.total_cycle);
    latency_stats_init(&g_latency_stats.loop_time);
}

// ==================== THROUGHPUT & TX STATISTICS ====================
typedef struct {
    uint32_t total_tx_attempts;
    uint32_t total_tx_success;
    uint32_t total_tx_failed;
    uint32_t total_bytes_tx;
    uint32_t start_time_ms;
} TxStats;

TxStats g_tx_stats;

void tx_stats_init() {
    g_tx_stats.total_tx_attempts = 0;
    g_tx_stats.total_tx_success = 0;
    g_tx_stats.total_tx_failed = 0;
    g_tx_stats.total_bytes_tx = 0;
    g_tx_stats.start_time_ms = millis();
}

void tx_stats_success(uint32_t bytes) {
    g_tx_stats.total_tx_attempts++;
    g_tx_stats.total_tx_success++;
    g_tx_stats.total_bytes_tx += bytes;
}

void tx_stats_failed() {
    g_tx_stats.total_tx_attempts++;
    g_tx_stats.total_tx_failed++;
}

// ==================== SENSOR STATISTICS ====================
typedef struct {
    uint32_t total_reads;
    uint32_t read_errors;
    uint32_t samples_dropped;   // Due to encryption/TX busy
} SensorStats;

SensorStats g_sensor_stats;

void sensor_stats_init() {
    g_sensor_stats.total_reads = 0;
    g_sensor_stats.read_errors = 0;
    g_sensor_stats.samples_dropped = 0;
}

// ==================== TASK LOGGER (Raw Data) ====================
typedef struct {
    uint16_t sensor[LOG_SIZE];
    uint16_t encrypt[LOG_SIZE];
    uint16_t lora_tx[LOG_SIZE];
    uint16_t total_cycle[LOG_SIZE];

    uint16_t idx_sensor;
    uint16_t idx_encrypt;
    uint16_t idx_lora;
    uint16_t idx_cycle;

    bool printed;
} TaskLogger;

TaskLogger logger = {0};

void log_task(uint8_t task_id, uint16_t value) {
    if (logger.printed) return;

    switch (task_id) {
        case 0:
            if (logger.idx_sensor < LOG_SIZE) logger.sensor[logger.idx_sensor++] = value;
            break;
        case 1:
            if (logger.idx_encrypt < LOG_SIZE) logger.encrypt[logger.idx_encrypt++] = value;
            break;
        case 2:
            if (logger.idx_lora < LOG_SIZE) logger.lora_tx[logger.idx_lora++] = value;
            break;
        case 3:
            if (logger.idx_cycle < LOG_SIZE) logger.total_cycle[logger.idx_cycle++] = value;
            break;
    }
}

void dump_logs() {
    Serial.println("\n===== RAW DATA LOGS FOR ANALYSIS =====\n");

    Serial.println("SENSOR_LOG_START");
    for (int i = 0; i < logger.idx_sensor; i++) Serial.println(logger.sensor[i]);
    Serial.println("SENSOR_LOG_END");

    Serial.println("ENCRYPT_LOG_START");
    for (int i = 0; i < logger.idx_encrypt; i++) Serial.println(logger.encrypt[i]);
    Serial.println("ENCRYPT_LOG_END");

    Serial.println("LORA_TX_LOG_START");
    for (int i = 0; i < logger.idx_lora; i++) Serial.println(logger.lora_tx[i]);
    Serial.println("LORA_TX_LOG_END");

    Serial.println("TOTAL_CYCLE_LOG_START");
    for (int i = 0; i < logger.idx_cycle; i++) Serial.println(logger.total_cycle[i]);
    Serial.println("TOTAL_CYCLE_LOG_END");

    logger.printed = true;
}

// ==================== COMPREHENSIVE REPORT ====================
void print_full_statistics_report() {
    uint32_t elapsed_ms = millis() - g_tx_stats.start_time_ms;
    float elapsed_sec = elapsed_ms / 1000.0f;

    Serial.println("\n");
    Serial.println("╔══════════════════════════════════════════════════════════════╗");
    Serial.println("║      SENDER NODE - COMPREHENSIVE STATISTICS REPORT           ║");
    Serial.println("╠══════════════════════════════════════════════════════════════╣");
    Serial.printf("║  Node ID       : %d\n", u8SlaveNodeID);
    Serial.printf("║  Test Duration : %.2f seconds\n", elapsed_sec);
    Serial.printf("║  TX Interval   : %d ms\n", TX_INTERVAL_MS);
    Serial.println("╠══════════════════════════════════════════════════════════════╣");

    // ===== 1. TX STATISTICS =====
    Serial.println("║                                                              ║");
    Serial.println("║  [1] TRANSMISSION STATISTICS                                 ║");
    Serial.println("║  ─────────────────────────────────────────────────────────── ║");

    float tx_success_rate = (g_tx_stats.total_tx_attempts > 0)
        ? ((float)g_tx_stats.total_tx_success / g_tx_stats.total_tx_attempts * 100.0f)
        : 0.0f;

    Serial.printf("║  Total TX Attempts  : %lu\n", g_tx_stats.total_tx_attempts);
    Serial.printf("║  TX Success         : %lu\n", g_tx_stats.total_tx_success);
    Serial.printf("║  TX Failed          : %lu\n", g_tx_stats.total_tx_failed);
    Serial.printf("║  TX Success Rate    : %.2f%%\n", tx_success_rate);
    Serial.printf("║  Total Bytes TX     : %lu\n", g_tx_stats.total_bytes_tx);
    Serial.printf("║  Current Seq ID     : %u\n", g_seq_id);

    // ===== 2. THROUGHPUT =====
    Serial.println("║                                                              ║");
    Serial.println("║  [2] THROUGHPUT                                              ║");
    Serial.println("║  ─────────────────────────────────────────────────────────── ║");

    float tx_pps = (elapsed_sec > 0) ? (g_tx_stats.total_tx_success / elapsed_sec) : 0;
    float tx_bps = (elapsed_sec > 0) ? ((g_tx_stats.total_bytes_tx * 8) / elapsed_sec) : 0;

    Serial.printf("║  TX Rate            : %.2f packets/sec\n", tx_pps);
    Serial.printf("║  TX Bit Rate        : %.2f bps (%.2f kbps)\n", tx_bps, tx_bps / 1000.0f);

    // ===== 3. SENSOR STATISTICS =====
    Serial.println("║                                                              ║");
    Serial.println("║  [3] SENSOR STATISTICS                                       ║");
    Serial.println("║  ─────────────────────────────────────────────────────────── ║");

    float sensor_success_rate = (g_sensor_stats.total_reads > 0)
        ? ((float)(g_sensor_stats.total_reads - g_sensor_stats.read_errors) / g_sensor_stats.total_reads * 100.0f)
        : 0.0f;

    Serial.printf("║  Total Reads        : %lu\n", g_sensor_stats.total_reads);
    Serial.printf("║  Read Errors        : %lu\n", g_sensor_stats.read_errors);
    Serial.printf("║  Samples Dropped    : %lu\n", g_sensor_stats.samples_dropped);
    Serial.printf("║  Sensor Success Rate: %.2f%%\n", sensor_success_rate);

    // ===== 4. LATENCY STATISTICS =====
    Serial.println("║                                                              ║");
    Serial.println("║  [4] LATENCY STATISTICS (microseconds)                       ║");
    Serial.println("║  ─────────────────────────────────────────────────────────── ║");

    // Sensor Read
    LatencyStats* sensor = &g_latency_stats.sensor_read;
    Serial.println("║  >> Sensor Read:");
    Serial.printf("║     Count: %lu | Min: %.0f | Max: %.0f | Mean: %.0f\n",
                  sensor->count, sensor->min_us, sensor->max_us, latency_stats_mean(sensor));
    Serial.printf("║     Std: %.2f | Jitter: %.2f\n",
                  latency_stats_std(sensor), latency_stats_jitter(sensor));

    Serial.printf("║     Count: %lu | Min: %.0f | Max: %.0f | Mean: %.0f\n",
                  encrypt->count, encrypt->min_us, encrypt->max_us, latency_stats_mean(encrypt));
    Serial.printf("║     Std: %.2f | Jitter: %.2f\n",
                  latency_stats_std(encrypt), latency_stats_jitter(encrypt));

    // LoRa TX
    LatencyStats* lora = &g_latency_stats.lora_tx;
    Serial.println("║  >> LoRa TX:");
    Serial.printf("║     Count: %lu | Min: %.0f | Max: %.0f | Mean: %.0f\n",

    Serial.printf("║     Std: %.2f | Jitter: %.2f\n",
                  latency_stats_std(lora), latency_stats_jitter(lora));

    LatencyStats* cycle = &g_latency_stats.total_cycle;
    Serial.println("║  >> Total Cycle (Sensor -> TX Done):");
    Serial.printf("║     Count: %lu | Min: %.0f | Max: %.0f | Mean: %.0f\n",
                  cycle->count, cycle->min_us, cycle->max_us, latency_stats_mean(cycle));
    Serial.printf("║     Std: %.2f | Jitter: %.2f\n",
                  latency_stats_std(cycle), latency_stats_jitter(cycle));

    // ===== 5. LOOP TIME =====
    Serial.println("║                                                              ║");
    Serial.println("║  [5] LOOP TIME                                               ║");
    Serial.println("║  ─────────────────────────────────────────────────────────── ║");

    LatencyStats* loop_t = &g_latency_stats.loop_time;
    Serial.printf("║  Total Loops  : %lu\n", loop_t->count);
    Serial.printf("║  Min          : %.0f us\n", loop_t->min_us);
    Serial.printf("║  Max          : %.0f us\n", loop_t->max_us);
    Serial.printf("║  Mean         : %.2f us\n", latency_stats_mean(loop_t));
    Serial.printf("║  Std Dev      : %.2f us\n", latency_stats_std(loop_t));
    Serial.printf("║  Jitter       : %.2f us\n", latency_stats_jitter(loop_t));

    float loop_freq = (latency_stats_mean(loop_t) > 0) ? (1000000.0f / latency_stats_mean(loop_t)) : 0;
    Serial.printf("║  Loop Freq    : %.2f Hz\n", loop_freq);

    Serial.println("║                                                              ║");
    Serial.println("╚══════════════════════════════════════════════════════════════╝");
    Serial.println("\n");
}

// ==================== GLOBAL VARIABLES ====================
SX1278 radio = new Module(ss, dio0, rst);

// Buffer for TX
uint8_t g_tx_buffer[PACKET_TOTAL_LEN];
uint8_t g_cipher_buffer[SECURE_DATA_TOTAL_LEN];

// Latest sensor sample
IMUSample g_current_sample;
bool g_sample_ready = false;

// Timing
uint32_t lastTxTime = 0;
uint32_t lastSensorRead = 0;
uint32_t lastStatsReport = 0;

// TX state
volatile bool txDoneFlag = false;
volatile bool txBusy = false;

// ==================== ISR ====================
void IRAM_ATTR dio0_isr() {
    txDoneFlag = true;
}

// ==================== NODE ID DETECTION ====================
void detect_node_id() {
    uint64_t chipid = ESP.getEfuseMac();
    Serial.printf("[MAIN] ESP32 Chip ID: %llX\n", chipid);

    if (chipid == 0x34B7C9EA6BE8ULL) {
        u8SlaveNodeID = 1;
    } else if (chipid == 0x5C9D26077000ULL) {
        u8SlaveNodeID = 2;
    } else if (chipid == 0x5C07B26062ECULL) {
        u8SlaveNodeID = 3;
    } else {
        u8SlaveNodeID = 0;  // Unknown node
    }

    Serial.printf("[MAIN] Assigned Node ID: %d\n", u8SlaveNodeID);
}

// ==================== SENSOR FUNCTIONS ====================
bool read_sensor(IMUSample* sample) {
    uint32_t start = micros();

    g_sensor_stats.total_reads++;

    int result = sensor_read(sample);

    uint32_t end = micros();
    uint32_t latency = end - start;

    if (result == 0) {
        sample->id = u8SlaveNodeID;
        latency_stats_update(&g_latency_stats.sensor_read, (float)latency);
        log_task(0, latency);
        return true;
    } else {
        g_sensor_stats.read_errors++;
        return false;
    }
}

// ==================== ENCRYPT FUNCTION ====================
bool encrypt_sample(const IMUSample* sample, uint8_t* cipher_out) {
    uint32_t start = micros();

    bool result = secure_data_encrypt(*sample, cipher_out);

    uint32_t end = micros();
    uint32_t latency = end - start;

    if (result) {
        latency_stats_update(&g_latency_stats.encrypt, (float)latency);
        log_task(1, latency);
    }

    return result;
}

// ==================== LORA TX FUNCTION ====================
bool transmit_packet(uint8_t* buffer, size_t len) {
    uint32_t start = micros();

    // Configure for uplink
    radio_config_uplink();

    // Transmit
    int state = radio.transmit(buffer, len);

    uint32_t end = micros();
    uint32_t latency = end - start;

    if (state == RADIOLIB_ERR_NONE) {
        latency_stats_update(&g_latency_stats.lora_tx, (float)latency);
        log_task(2, latency);
        tx_stats_success(len);
        return true;
    } else {
        tx_stats_failed();
        Serial.printf("[LORA] TX failed, code: %d\n", state);
        return false;
    }
}

// ==================== SETUP ====================
void setup() {
    Serial.begin(115200);
    Serial.println("\n");
    Serial.println("╔══════════════════════════════════════════════════════════════╗");
    Serial.println("║      NON-RTOS, NON-TDMA SENDER - BENCHMARK MODE              ║");
    Serial.println("╚══════════════════════════════════════════════════════════════╝");
    Serial.println();

    // Detect and set node ID
    detect_node_id();

    // Initialize statistics
    system_latency_init();
    tx_stats_init();
    sensor_stats_init();

    // Initialize I2C mutex (not used in non-RTOS but keep for compatibility)
    // gI2CMutex = xSemaphoreCreateMutex();

    // Initialize LoRa
    InitLora();
    radio.setPacketSentAction(dio0_isr);

    // Initialize MPU6050
    Init_MPU6050();

    Serial.println("[MAIN] Setup complete - Polling mode active");
    Serial.printf("[MAIN] TX Interval: %d ms\n\n", TX_INTERVAL_MS);

    // Initialize timing
    lastTxTime = millis();
    lastSensorRead = millis();
    lastStatsReport = millis();
}

// ==================== MAIN LOOP ====================
void loop() {
    uint32_t loop_start = micros();
    uint32_t now = millis();

    // ========== 1. SENSOR READING (continuous, latest value) ==========
    if (now - lastSensorRead >= SENSOR_SAMPLE_MS) {
        lastSensorRead = now;

        IMUSample sample;
        if (read_sensor(&sample)) {
            // Store latest sample
            memcpy(&g_current_sample, &sample, sizeof(IMUSample));
            g_sample_ready = true;
        }
    }

    // ========== 2. TRANSMIT AT FIXED INTERVAL ==========
    if (now - lastTxTime >= TX_INTERVAL_MS) {
        lastTxTime = now;

        if (g_sample_ready) {
            uint32_t cycle_start = micros();

            // Step 1: Encrypt
            if (encrypt_sample(&g_current_sample, g_cipher_buffer)) {

                // Step 2: Build TX packet (cipher + seq_id)
                memcpy(g_tx_buffer, g_cipher_buffer, SECURE_DATA_TOTAL_LEN);
                memcpy(g_tx_buffer + SECURE_DATA_TOTAL_LEN, &g_seq_id, sizeof(g_seq_id));

                // Step 3: Transmit
                if (transmit_packet(g_tx_buffer, PACKET_TOTAL_LEN)) {
                    g_seq_id++;

                    #if DEBUG_APP == 1
                    Serial.printf("[TX] seq=%u, Node=%d\n", g_seq_id - 1, u8SlaveNodeID);
                    #endif
                }
            } else {
                Serial.println("[MAIN] Encryption failed");
            }

            uint32_t cycle_end = micros();
            uint32_t cycle_latency = cycle_end - cycle_start;
            latency_stats_update(&g_latency_stats.total_cycle, (float)cycle_latency);
            log_task(3, cycle_latency);

            g_sample_ready = false;  // Mark as consumed
        } else {
            g_sensor_stats.samples_dropped++;
        }
    }

    // ========== 3. PERIODIC STATISTICS REPORT ==========
    if (now - lastStatsReport >= 30000) {
        lastStatsReport = now;
        print_full_statistics_report();
    }

    // ========== 4. DUMP RAW LOGS (once when buffers full) ==========
    if (!logger.printed &&
        logger.idx_sensor >= LOG_SIZE &&
        logger.idx_encrypt >= LOG_SIZE &&
        logger.idx_lora >= LOG_SIZE) {
        dump_logs();
    }

    // ========== 5. LOG LOOP TIME ==========
    uint32_t loop_end = micros();
    uint32_t loop_latency = loop_end - loop_start;
    latency_stats_update(&g_latency_stats.loop_time, (float)loop_latency);

    // Small delay to prevent watchdog (and control loop rate)
    delay(1);
}
